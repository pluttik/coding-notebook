Fixtures in JUnit5

// Fixtures are declared inside a test class and build a known and fixed environment so that test results are repeatable.

@BeforeAll // run once before all tests

@BeforeEach // run before each test

@Test // test

@AfterEach // run after each test

@AfterAll // run once after all tests

// N.B. @BeforeAll and @AfterAll are static and run before/after instantiation of test class.


Assertions in JUnit5

import static org.junit.jupiter.api.Assertions.*;

assertEquals(expected, actual, optional double delta, optional message)

// compares using .equals()

// the double delta is for specifying precision when comparing expected and actual doubles

assertSame(expected, actual, optional message)

// compares using ==

assertTrue(boolean condition)

assertFalse(boolean condition)
