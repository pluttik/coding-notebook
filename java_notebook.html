Inheritance

//make a parent class

public class MyString {
protected int myInt; //protected means only available to its children
public MyString (String myString) { //constructor
this.myString = myString;
}
}


//make a child class that inherits

public class MyNumber extends SomeClass {
public int myInt;

public MyNumber (String myString, int myInt) { //constructor
super(myString);
this.myInt = myInt;
}
